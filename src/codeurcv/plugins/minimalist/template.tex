% Custom commands
% \newcommand{\resumeItem}[2]{

\newcommand{\resumeItem}[1]{
  \item\small{
    {#1 \vspace{-2pt}}
  }
}

% Just in case someone needs a heading that does not need to be in a list
\newcommand{\resumeHeading}[4]{
    \begin{tabular*}{0.99\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-5pt}
}

\newcommand{\resumeSubheading}[4]{
  \vspace{-2pt}\item
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeSubSubheading}[2]{
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-7pt}
}

\newcommand{\resumeSubItem}[1]{\resumeItem{#1}\vspace{-4pt}}

\renewcommand{\labelitemii}{$\circ$}

\newcommand{\resumeSubHeadingListStart}{\begin{itemize}[leftmargin=0.15in, label={}]}
\newcommand{\resumeSubHeadingListEnd}{\end{itemize}}
\newcommand{\resumeItemListStart}{\begin{itemize}}
\newcommand{\resumeItemListEnd}{\end{itemize}\vspace{-5pt}}

\newcommand{\technologies}[1]{
\noindent\textbf{Technologies:} #1
}

\newcommand{\resumeProjectHeading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \small#1 & #2 \\
    \end{tabular*}\vspace{-7pt}
}

    \href{tel:\VAR{basic_details.phone}}{\VAR{basic_details.phone}} $|$ 
    \href{\VAR{basic_details.github}}{\VAR{basic_details.github}} $|$ 
    \href{\VAR{basic_details.linkedin}}{\VAR{basic_details.linkedin}}

\end{center}

%----------SUMMARY-----------------
\section{Summary}
\resumeSubHeadingListStart
  \item\small
    {\VAR{summary}}
\resumeSubHeadingListEnd

%-----------EDUCATION-----------------
\section{Education}
  \resumeSubHeadingListStart
  \BLOCK{for school in education}
    \resumeSubheading
      { {\VAR{school.institution}} }{ {\VAR{school.location}} }
      { {\VAR{school.degree}}; GPA: {\VAR{school.gpa}} }{ {\VAR{school.year | string}} }
    \BLOCK{endfor}
  \resumeSubHeadingListEnd

%-----------EXPERIENCE-----------------
\section{Experience}
  \resumeSubHeadingListStart
  \BLOCK{for job in work}
    \resumeSubheading
      { {\VAR{job.company}} }{ {\VAR{job.location}} }
      { {\VAR{job.role}} }{ {\VAR{job.start}} - {\VAR{job.end}} }
      \resumeItemListStart
      \BLOCK{for item in job.achievements}
            \resumeItem{ {\VAR{item | safe}} }
      \BLOCK{endfor}
      \resumeItem{ \technologies{\VAR{job.technologies | join(", ")}} }
      \resumeItemListEnd
  \BLOCK{endfor}
\resumeSubHeadingListEnd


%-----------PROJECTS-----------------
\section{Projects}
\resumeSubHeadingListStart
\BLOCK{for proj in projects}
  \resumeProjectHeading
    {\href{ {\VAR{proj.link}} }{ \textbf{\VAR{proj.name}} }((* if proj.technologies *)) $|$ \emph{{\VAR{proj.technologies | join(', ')}}}((* endif *))}
    { {\VAR{proj.start}} - {\VAR{proj.end}} }
  \resumeItemListStart
    \BLOCK{for item in proj.description}
    \resumeItem{ {\VAR{item | safe}} }
    \BLOCK{endfor}
  \resumeItemListEnd
\BLOCK{endfor}
\resumeSubHeadingListEnd

%-----------TECHNICAL SKILLS-----------------
\section{Technical Skills}
\resumeSubHeadingListStart
\BLOCK{for skill in skills}
  \item\small 
  \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r} 
    \textbf{ {\VAR{skill.category|capitalize}} }: { {\VAR{skill.featured | join(", ")}} } \\
  \end{tabular*}\vspace{-7pt}
\BLOCK{endfor}
\resumeSubHeadingListEnd
